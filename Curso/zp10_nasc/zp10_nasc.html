data: v_ndias TYPE i.

PARAMETERS: p_nasc TYPE sy-datum.

*--------------------------

START-OF-SELECTION.

call function 'ZMF10_CALC_DATA'
  exporting
    datai         = P_NASC
    dataf         = sy-datum
 IMPORTING
   NDIAS         = v_ndias
 EXCEPTIONS
   EDATA1        = 1
   EDATA2        = 2
   EDATA3        = 3
   OTHERS        = 4.

  case sy-subrc.
    when 0.
      MESSAGE i000 with 'Voce viveu:' v_ndias 'dias'.
    when 1.
        WRITE: / 'Data de nasvimento, não foi informado'.
    WHEN 3.
         WRITE: / 'Voce ainda não nasceu'.

    when OTHERS.
      ENDCASE.

end-of-SELECTION.
